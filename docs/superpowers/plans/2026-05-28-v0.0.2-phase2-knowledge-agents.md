# v0.0.2 Phase 2: Knowledge Base + Agent Configuration — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 建立分类知识库体系（法律法规库、企业模板库、风险项库、自定义库）+ 智能体配置界面（知识库关联、规则集、参数调整），为 Phase 3 的业务流程提供知识支撑。

**Architecture:** 在 Phase 1 的模型管理基础上，新增 4 张知识库核心表 + 1 张规则集表。知识库文档通过 pgvector 向量化，智能体调用时检索。所有知识库按组织隔离，支持多用户协作积累。

**Tech Stack:** Next.js 15, Drizzle ORM, PostgreSQL + pgvector, Mastra RAG

**Spec:** `docs/superpowers/specs/2026-05-28-v0.0.2-platform-design.md` Section 六

**Depends on:** Phase 1（agentDefinitions 表、组织隔离）

- [ ] **Step 1: 创建知识库核心表**

```typescript
// 知识库类型枚举
export const knowledgeBaseTypeEnum = pgEnum("kb_type", [
  "legal_regulation",  // 法律法规库
  "bid_template",      // 企业模板库
  "risk_item",         // 风险项库
  "custom",            // 自定义库
]);

// 知识库表
export const knowledgeBases = pgTable("knowledge_bases", {
  id: uuid("id").primaryKey().defaultRandom(),
  name: varchar("name", { length: 255 }).notNull(),
  type: knowledgeBaseTypeEnum("type").notNull(),
  description: text("description"),
  icon: varchar("icon", { length: 50 }),
  organizationId: uuid("organization_id").notNull(),
  isActive: boolean("is_active").default(true),
  documentCount: integer("document_count").default(0),
  totalChunks: integer("total_chunks").default(0),
  createdAt: timestamp("created_at", { mode: "date" }).defaultNow(),
  updatedAt: timestamp("updated_at", { mode: "date" }).defaultNow(),
});

// 知识条目表（通用，存储各类知识内容）
export const knowledgeItems = pgTable("knowledge_items", {
  id: uuid("id").primaryKey().defaultRandom(),
  knowledgeBaseId: uuid("knowledge_base_id").notNull().references(() => knowledgeBases.id, { onDelete: "cascade" }),
  title: varchar("title", { length: 500 }).notNull(),
  content: text("content").notNull(),
  source: varchar("source", { length: 255 }),             // 来源标识
  metadata: jsonb("metadata"),                              // 类型特定字段（见下方）
  tags: jsonb("tags").$type<string[]>().default([]),
  isVectorized: boolean("is_vectorized").default(false),
  chunkCount: integer("chunk_count").default(0),
  createdBy: uuid("created_by"),
  createdAt: timestamp("created_at", { mode: "date" }).defaultNow(),
  updatedAt: timestamp("updated_at", { mode: "date" }).defaultNow(),
});

// 知识条目向量块表（分块 + 向量化）
export const knowledgeItemChunks = pgTable("knowledge_item_chunks", {
  id: uuid("id").primaryKey().defaultRandom(),
  itemId: uuid("item_id").notNull().references(() => knowledgeItems.id, { onDelete: "cascade" }),
  chunkIndex: integer("chunk_index").notNull(),
  content: text("content").notNull(),
  embedding: vector("embedding", { dimensions: 1536 }),     // OpenAI ada-002 维度
  tokenCount: integer("token_count"),
  createdAt: timestamp("created_at", { mode: "date" }).defaultNow(),
});
```

`metadata` 字段按知识库类型存储不同结构：

```
法律法规库 metadata: {
  lawType: "法律" | "行政法规" | "部门规章" | "地方性法规" | "国标",
  documentNumber: "国务院令第613号",
  publishDate: "2024-01-01",
  effectiveDate: "2024-07-01",
  isLatestVersion: true,
  supersededBy: null | "新版文号",
  chapters: [{ title: "第一章 总则", clauses: [...] }]
}

企业模板库 metadata: {
  industry: "建筑" | "市政" | "公路" | "水利" | "电力" | ...,
  templateType: "技术标" | "商务标" | "资信标" | "投标函" | "综合标",
  outline: [{ sectionNo: "1", title: "...", parentId: null, required: true, children: [...] }],
  sourceProjectId: "uuid" | null
}

风险项库 metadata: {
  riskType: "废标" | "格式" | "合规" | "业绩" | "资质" | "技术",
  detectionRule: {
    type: "keyword" | "comparison" | "semantic" | "existence",
    logic: "自然语言描述的检测逻辑",
    parameters: { key: value }
  },
  severity: "high" | "medium" | "low",
  legalReference: "关联的法律法规条目 ID",
  suggestion: "标准修改建议文本",
  frequency: 0  // 出现频次，自动统计
}
```

```typescript
// 规则检测类型
export const ruleDetectionTypeEnum = pgEnum("rule_detection_type", [
  "keyword",      // 关键词匹配
  "comparison",   // 数值比对
  "semantic",     // 语义比对
  "existence",    // 存在性检查
]);

// 规则集表
export const ruleSets = pgTable("rule_sets", {
  id: uuid("id").primaryKey().defaultRandom(),
  name: varchar("name", { length: 255 }).notNull(),
  description: text("description"),
  industry: varchar("industry", { length: 100 }),           // 适用行业
  agentId: uuid("agent_id").references(() => require("./models").agentDefinitions.id),
  isActive: boolean("is_active").default(true),
  organizationId: uuid("organization_id"),
  createdAt: timestamp("created_at", { mode: "date" }).defaultNow(),
  updatedAt: timestamp("updated_at", { mode: "date" }).defaultNow(),
});

// 规则条目表
export const ruleItems = pgTable("rule_items", {
  id: uuid("id").primaryKey().defaultRandom(),
  ruleSetId: uuid("rule_set_id").notNull().references(() => ruleSets.id, { onDelete: "cascade" }),
  ruleNo: varchar("rule_no", { length: 20 }).notNull(),     // R001, R002...
  name: varchar("name", { length: 255 }).notNull(),
  detectionType: ruleDetectionTypeEnum("detection_type").notNull(),
  severity: varchar("severity", { length: 10 }).notNull(),   // high / medium / low
  description: text("description").notNull(),                 // 检测逻辑自然语言描述
  parameters: jsonb("parameters"),                            // 检测参数
  isEnabled: boolean("is_enabled").default(true),
  sortOrder: integer("sort_order").default(0),
  createdAt: timestamp("created_at", { mode: "date" }).defaultNow(),
});
```

```typescript
export class EmbeddingService {
  private chunkSize = 500;    // 每块最大 token 数
  private chunkOverlap = 50;  // 块间重叠

  // 将长文本分块
  chunkText(text: string): string[];

  // 对单个块生成 embedding（优先本地模型，回退云端 API）
  async embed(chunk: string): Promise<number[]>;

  // 批量向量化一个知识条目
  async vectorizeItem(itemId: string): Promise<{ chunks: number }>;

  // 语义搜索

}
```
