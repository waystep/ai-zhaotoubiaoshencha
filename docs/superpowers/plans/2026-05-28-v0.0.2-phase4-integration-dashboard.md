```typescript
  "analysis.completed",     // A1 招标解析完成
  "draft.generated",        // A2 投标样稿生成完成
  "review.completed",       // A3 投标预审完成

]);

export const webhooks = pgTable("webhooks", {
  id: uuid("id").primaryKey().defaultRandom(),
  name: varchar("name", { length: 255 }).notNull(),
  url: text("url").notNull(),
  secret: varchar("secret", { length: 255 }),          // HMAC 签名密钥
  events: jsonb("events").$type<string[]>().notNull(),  // 订阅的事件列表
  headers: jsonb("headers"),                             // 自定义请求头
  isActive: boolean("is_active").default(true),
  retryCount: integer("retry_count").default(3),
  organizationId: uuid("organization_id"),
  createdBy: uuid("created_by"),
  createdAt: timestamp("created_at", { mode: "date" }).defaultNow(),
  updatedAt: timestamp("updated_at", { mode: "date" }).defaultNow(),
});

export const webhookDeliveryLogs = pgTable("webhook_delivery_logs", {
  id: uuid("id").primaryKey().defaultRandom(),
  webhookId: uuid("webhook_id").notNull().references(() => webhooks.id),
  event: webhookEvents("event").notNull(),
  projectId: uuid("project_id"),
  requestPayload: jsonb("request_payload"),
  responseStatus: integer("response_status"),
  responseBody: text("response_body"),
  durationMs: integer("duration_ms"),
  success: boolean("success"),
  attemptCount: integer("attempt_count").default(1),
  createdAt: timestamp("created_at", { mode: "date" }).defaultNow(),
});
```

- [ ] **Step 1: 实现 WebhookDispatcher**

```typescript
export class WebhookDispatcher {
  // 事件触发时调用：查找订阅了该事件的所有 webhook → 逐一推送
  async dispatch(event: WebhookEvent, projectId: string, payload: object): Promise<void>;

  // 单个 webhook 推送（带重试）
  private async deliver(webhook: Webhook, payload: object): Promise<DeliveryResult>;

  // HMAC 签名
  private sign(payload: string, secret: string): string;

  // 重试逻辑：指数退避，最多 retryCount 次
  private async retryWithBackoff(webhook: Webhook, payload: object): Promise<void>;
}
```

在 Phase 3 的智能体完成回调中注入 dispatch 调用：
- A1 完成 → `dispatch("analysis.completed", projectId, analysisResult)`
- A3 完成 → `dispatch("review.completed", projectId, riskList)`
- A6 完成 → `dispatch("report.completed", projectId, report)`

```typescript
export class DashboardService {
  // 核心指标
  async getCoreMetrics(orgId: string): Promise<{
    totalProjects: number;
    totalTenderDocs: number;
    totalBidDocs: number;
    avgReviewScore: number;
    totalRisks: number;
    risksBySeverity: { high: number; medium: number; low: number };
    bidRejectionAvoidanceRate: number;
    monthOverMonthChanges: Record<string, number>;
  }>;

  // 招标数据分析
  async getTenderAnalytics(orgId: string, period: "month" | "quarter" | "year"): Promise<{
    industryDistribution: { name: string; count: number }[];
    amountDistribution: { range: string; count: number }[];
    evaluationMethodDistribution: { method: string; count: number }[];
    qualificationTop5: { name: string; count: number }[];
    monthlyTrend: { month: string; tenderCount: number; parsedCount: number }[];
  }>;

  // 投标数据分析
  async getBidAnalytics(orgId: string): Promise<{
    generationVsReviewTrend: { month: string; generated: number; reviewed: number }[];
    sourceDistribution: { source: string; count: number }[];
    editRate: number;
    scoreDistribution: { range: string; count: number }[];
    statusFunnel: { stage: string; count: number }[];
    avgProcessingTime: { agentKey: string; avgMs: number }[];
  }>;

  // 风险洞察
  async getRiskInsights(orgId: string): Promise<{
    severityOverview: { high: number; medium: number; low: number; highPercentage: number; monthChange: number };
    riskTypeTop10: { type: string; count: number }[];
    highFrequencyRisks: { name: string; count: number; trend: "up" | "down" | "stable" }[];
    riskTrend: { month: string; highRiskCount: number; avgPerProject: number; avoidanceRate: number }[];
    legalVerificationStats: { verified: number; outdated: number; notFound: number };
  }>;

  // 知识沉淀
  async getKnowledgeStats(orgId: string): Promise<{
    baseStats: { type: string; itemCount: number; vectorizedCount: number }[];
    lawCoverageRate: number;
    templateUsageTop5: { name: string; count: number }[];
    riskItemGrowth: number;
    contributorRanking: { userId: string; name: string; contributions: number }[];
  }>;

  // 模型用量
  async getModelUsage(orgId: string): Promise<{
    localVsCloud: { local: number; cloud: number };
    modelCallRanking: { modelName: string; callCount: number }[];
    tokenTrend: { date: string; inputTokens: number; outputTokens: number }[];
    estimatedCost: number;
    modelSuccessRates: { modelName: string; success: number; failed: number }[];
  }>;

  // 团队动态
  async getTeamActivity(orgId: string): Promise<{
    activeUserRanking: { userId: string; name: string; operations: number }[];
    operationHeatmap: { hour: number; day: number; count: number }[];
    recentOperations: { userId: string; action: string; timestamp: Date }[];
    departmentComparison: { department: string; projects: number; reviews: number; risks: number }[];
  }>;
}
```

每个方法内部使用 Drizzle 的聚合查询（count, avg, groupBy）+ 原始 SQL（复杂统计）。
